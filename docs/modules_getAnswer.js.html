modules/getAnswer.js

'use strict'
const sqlite = require('sqlite-async')
/**
 * @module
 */

/**
 * Fetch all answers for a question from the database.
 * @param {string} database The database file path
 * @param {integer} id The unique question ID to find answers for
 * @returns {object} All answers for the specific question
 */
module.exports = async(database, id) => {
	try {
		const db = await sqlite.open(database)
		const sql = `SELECT * FROM Answers WHERE questionID = ${id} ORDER BY correct DESC;`
		const answerData = await db.all(sql)
		await db.close()
		return answerData
	} catch (err) {
		throw err
	}
}
